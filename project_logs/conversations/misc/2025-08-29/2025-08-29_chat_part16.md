#!/usr/bin/env python3
import sys, yaml, json, datetime, os
from collections import defaultdict

# Usage:
#   python tools/assess_eval.py assessments/blueprints/grade10/answer_key.yaml '{"Q1":"C","Q2":"D",...}'
# or provide answers via a JSON file path.

def load_answers(arg):
    if os.path.isfile(arg):
        with open(arg) as f: return json.load(f)
    return json.loads(arg)

def main():
    if len(sys.argv) < 3:
        print("Usage: assess_eval.py <answer_key.yaml> <answers.json|inline_json> [student_name]")
        sys.exit(1)
    key_path = sys.argv[1]
    answers = load_answers(sys.argv[2])
    student_name = sys.argv[3] if len(sys.argv) > 3 else "Unknown"

    with open(key_path) as f:
        key = yaml.safe_load(f)

    # Gather items
    clusters = key["clusters"]
    total = 0; correct = 0
    strand_totals = defaultdict(int); strand_correct = defaultdict(int)
    for cl in clusters:
        strand = cl["strand"]
        for item in cl["items"]:
            qid = item["id"].replace(";","").strip()
            k = item["key"].strip().upper()
            total += 1
            strand_totals[strand] += 1
            a = answers.get(qid, "").strip().upper()
            if a == k:
                correct += 1
                strand_correct[strand] += 1

    overall = correct / total if total else 0.0

    # Load scoring rules
    rules_path = os.path.join(os.path.dirname(os.path.dirname(key_path)),'_shared','scoring_rules.yaml')
    with open(rules_path) as f:
        rules = yaml.safe_load(f)
    cutoff = float(rules.get("mastery_cutoff", 0.9))
    order = rules["placement_logic"]["order"]

    first_below = None
    for s in order:
        if strand_totals[s]:
            s_acc = strand_correct[s] / strand_totals[s]
            if s_acc < cutoff:
                first_below = (s, s_acc)
                break

    # Recommend starting worksheet path (vA variant)
    # Example mapping rule: use the cluster code for the first-below strand (e.g., B1 -> start series)
    start_path = ""
    if first_below:
        strand = first_below[0]
        # crude example mapping â€” refine with your real codes:
        # pick the first cluster code that matches the strand
        code = next((cl["code"] for cl in clusters if cl["strand"]==strand), "X0")
        # choose a generic skill name to be replaced later
        skill_name = "Placement_Start"
        start_path = f"materials/worksheets/math/gr10/{strand}/Gr10_Math_{strand}_{code}_{skill_name}_vA.pdf"

    # Build parent report data
    strengths = ", ".join([s for s in order if strand_totals[s] and (strand_correct[s]/strand_totals[s]) >= cutoff]) or "â€”"
    focus = first_below[0] if first_below else "â€”"
    strand_score = f"{round(first_below[1]*100)}%" if first_below else "â€”"

    print(json.dumps({
        "student_name": student_name,
        "date": datetime.date.today().isoformat(),
        "overall": round(overall*100),
        "correct": correct,
        "total": total,
        "first_below_strand": focus,
        "strand_score": strand_score,
        "start_path": start_path,
        "strand_breakdown": {
            s: {"correct": strand_correct[s], "total": strand_totals[s],
                "percent": round((strand_correct[s]/strand_totals[s])*100) if strand_totals[s] else 0}
            for s in order if strand_totals[s] or strand_correct[s]
        },
        "notes": {
            "cutoff": cutoff,
            "order": order
        }
    }, indent=2))
if __name__ == "__main__":
    main()
